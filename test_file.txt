def get_job_objects_one_page(r, all_jobs: List) -> None:
    soup = BeautifulSoup(r.text, "html.parser")

    current_job = None

    for step, i in enumerate(soup.find_all("td"), 1):
        if i:
            contents = i.getText()
        else:
            contents = "No"

        if step % 5 == 1:
            current_job = Job()

        if contents:
            if i["data-th"] == "Job Title":
                current_job.title = contents
            elif i["data-th"] == "Actions":
                current_job.actions = contents
            elif i["data-th"] == "Area of Interest":
                current_job.skill = contents
            elif i["data-th"] == "MyNewLocation1":
                current_job.location = contents
            elif i["data-th"] == "Alternate Location":
                current_job.alt_location = contents

        for link in i.find_all('a', attrs={'href': re.compile("^https://.*ProjectDetail.*")}):
            current_job.link = link.get('href')

        if step % 5 == 0:
            assert current_job, f"Current job is not defined, step: {step}"
            all_jobs.append(current_job)
